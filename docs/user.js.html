class User {
  constructor(userId) {
    this._userId = userId === undefined ? "" : userId;
    this._publicKey = "";
    this._privateKey = "";
  }

  get userId() {
    return this._userId;
  }

  get publicKey() {
    return this._publicKey;
  }

  set publicKey(publicKey = "") {
    this._publicKey = publicKey;
  }

  get privateKey() {
    return this._privateKey;
